Lark is an interpreted language that sits right in the object graph of a running Objective-C application. Lark bridges the functional paradigm of Scheme to Objective-C's naturally Erlang-ish message-passing semantics. It could only get better with macros! 


(define window [[[UIApplication sharedApplication] delegate] window])
(define setWindowColor
	(lambda (color)
		[window setBackgroundColor: [UIColor color]]))
(setWindowColor blueColor)

(define square (lambda (x) (* x x)))
(square 5) ;; => 25

(define makeSecretBox
	(lambda (key)
		[NSDictionary dictionaryWithObject: "secret" forKey: key]))
[(makeSecretBox "hidden-key!") objectForKey: "hidden-key!"] ;; => "secret"
